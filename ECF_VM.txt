/* ----------ANALYSE STRUCTURELLE 

La base de données contient 8 tables.

La table 'customers' contient les informations sur les clients de l'entreprise.
Chaque client est identifié via la clé principale 'customerNumber' (numéro de client).
Cette table est liée à la table 'employees' via la clé étrangère 'salesRepEmployeeNumber'(numéro de l'employé de vente associé à un client donné)

La table 'employees' contient les informations sur les employés de l'entreprise.
Chaque employé est identifié via la clé principale 'employeeNumber' (numéro d'employé).
Cette table est liée à la table 'offices' via la clé étrangère 'officeCode' (numéro de bureau au sein duquel travaille un employé donné),
ainsi qu'à elle-même via la clé étrangère 'reportsTo' (numéro d'employé du supérieur d'un employé donné, si existant)

La table 'offices' contient les informations sur les bureaux de l'entreprise.
Chaque bureau est identifé via la clé principale 'officeCode' (numéro de bureau).

La table 'orders' contient les informations sur les commandes passées auprès de l'entreprise.
Chaque commande est identifiée via la clé principale 'orderNumber' (numéro de commande).
Elle est liée à la table 'customers' via la clé étrangère 'customerNumber' (numéro du client ayant passé une commande donnée).

La table 'products' contient les informations sur les produits vendus par l'entreprise.
Chaque produit est indentifé via la clé principale 'productCode' (code produit).
Elle est liée à la table 'productlines' via la clé étrangère 'productLine' (numéro de la ligne de produit pour un produit donné).

La table 'productlines' contient les informations sur les lignes de produit proposées par l'entreprise.
Chaque ligne de produit est identifiée via la clé principale 'productLine' (numéro de ligne de produit).

La table 'orderdetails' contient le détail (produit par produit) des commandes passées auprès de l'entreprise.
Chaque détail de commande est identifée via le couple de clés principales ('orderNumber', 'productCode').
Cette table est liée à la table 'orders' via la clé étrangère 'orderNumber' (numéro de commande)
ainsi qu'à la table 'products' via la clé étrangère 'productCode' (code du produit).
La colonne 'orderLineNumber' (numéro de la ligne de produit pour un couple commande-produit donné) permet également un lien vers la table 'productlines'.

La table 'payments' contient les informations sur les paiements effectués à l'entreprise.
Chaque paiement est identifié via le couple de clés principales ('customerNumber', 'checkNumber').
Cette table est liée à la table 'customers' via la clé étrangère 'customerNumber' (numéro de client).

*/

/* ----------QUESTION 12
Dans la table des produits (products), nous voulons la quantité commandée, qui se
trouve dans les détails des ordres (orderDetails), totale, et la somme des ventes, depuis
le 01/01/2005, classées par ligne de produit (productLine).
*/

SELECT products.productCode 'Code du produit',
	products.productName 'Nom du produit',
	SUM(orderDetails.quantityOrdered) 'Quantité commandée',
	products.quantityInStock 'Quantité totale (en stock)',
	SUM(orderDetails.quantityOrdered*orderDetails.priceEach) 'Somme des ventes' 
FROM products
INNER JOIN orderDetails ON orderDetails.productCode = products.productCode
INNER JOIN orders ON orderDetails.orderNumber = orders.orderNumber
WHERE orders.orderDate > '2005-01-01'
GROUP BY products.productCode
ORDER BY products.productLine, products.productCode;

/* RESULTATS QUESTION 12
Empty set
(Il n'existe pas de commande en 2005 dans la table orders.)
*/

/* ----------QUESTION 13
Afficher les identifiants, les dates et statuts des commandes, le nom du client ayant
passé commande, ainsi que le nom, le code et la quantité d'articles commandé.
*/

SELECT orders.orderNumber 'Numéro de commande',
	orders.status 'Status de la commande',
	orders.orderDate 'Code de produit / Date achat',
	"-" AS 'Quantité achetée',
	customers.customerName 'Nom du produit / Nom du client'
	FROM orders	
	INNER JOIN customers ON orders.customerNumber = customers.customerNumber
UNION
SELECT orderDetails.orderNumber,
	"-",
	orderDetails.productCode,
	orderDetails.quantityOrdered,
	products.productName
	FROM orderDetails
	INNER JOIN products ON orderDetails.productCode = products.productCode
	WHERE orderDetails.orderNumber < 10164
ORDER BY 1, 2;

/* REMARQUE : les commandes numéro 10164 à 10223 ne figurent pas dans la table orders
bien que la table orderDetails contiennent les détails de ces commandes.
Etant donné qu'il est impossible de connaître le nom du client, le status et la date d'achat
de ces commandes, j'ai choisi de les exclure de ma requête SQL (WHERE orderDetails.orderNumber < 10164)

/* ----------QUESTION 14
À partir des tables customers, employees, offices, orders, orderDetails,
products afficher le nom du client, la ville et le pays du bureau, le prénom, le nom et
le poste occupé par l'employé, les articles commandés, leur quantité, la ligne de produit
et le prix d'achat.
*/

SELECT customers.customerNumber AS 'Numéro client', 
	customers.customerName AS 'Client', 
	offices.city AS 'Ville', 
	offices.country AS 'Pays', 
	employees.firstName AS 'Prénom employé', 
	employees.lastName AS 'Nom employé', 
	employees.jobTitle AS 'Poste occupé',
	'-' AS 'Articles commandés',
	'-' AS 'Quantité',
	'-' AS 'Ligne de produit',
	'-' AS 'Prix achat'
	FROM customers
	INNER JOIN employees ON customers.salesRepEmployeeNumber = employees.employeeNumber
	INNER JOIN offices ON employees.officeCode = offices.officeCode
UNION
SELECT orders.customerNumber, 
	'-', '-', '-', '-', '-', '-',
	products.productName, 
	SUM(orderDetails.quantityOrdered), 
	products.productLine, 
	products.buyPrice 
	FROM orderDetails
	INNER JOIN orders ON orderDetails.orderNumber = orders.orderNumber
	INNER JOIN products ON orderDetails.productCode = products.productCode
	GROUP BY orders.customerNumber, orderDetails.productCode
ORDER BY 1, 8;

/* REMARQUE : Comme mentionné précédemment, les commandes numéro 10164 à 10223 ne figurent pas
dans la table orders, ce qui rend leurs attributions au bon client. De ce fait, de nombreux clients
apparaissent comme n'ayant jamais commandé d'articles dans le résultat de la requête.
*/
